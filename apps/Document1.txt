TABLE FILE RPT_INST
PRINT RPT_NAME
BY INST_KEY
WHERE RPT_NAME CONTAINS '/' OR RPT_NAME CONTAINS '\'
END

JOIN SET_KEY IN RPT_SET TO ALL SET_KEY IN RPT_INST AS J1
TABLE FILE RPT_SET
PRINT RPT_NAME
BY ROLLUP_CODE
BY USER_NAME 
BY SET_NAME
BY INST_KEY
WHERE RPT_NAME CONTAINS '/' OR RPT_NAME CONTAINS '\'
END
